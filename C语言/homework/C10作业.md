```
#include <stdio.h>
#include <stdlib.h>

int count = 0;
void hanio(int n, char a, char b, char c){
    //把盘子移到另一个杆子上
    if(n == 1){//只有一个盘子，从a放到b上
        printf("%c-->%c\n",a,b);
        count++;
    }else{
        count++;
        hanio(n-1, a, c, b);//将a上盘子放到c上
        printf("%c-->%c\n",a,b);
        hanio(n-1, c, b, a);//将c上盘子放到a上
    }
}

int main(void){
    int n;
    printf("Input n:\n");
    scanf("%d",&n);
    hanio(n, 'A', 'B', 'C');
    printf("%d\n",count);
    return 0;
    
}
```
